local function getMyPlot()
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return nil end
    for _, plot in ipairs(plots:GetChildren()) do
        local channel = Synchronizer:Get(plot.Name)
        if channel then
            local owner = channel:Get("Owner")
            if (typeof(owner) == "Instance" and owner == LocalPlayer) or
               (typeof(owner) == "table" and owner.UserId == LocalPlayer.UserId) then
                return plot
            end
        end
    end
    return nil
end

local function getPetsOnPlot(plot)
    local pets = {}
    if not plot then return pets end
    local channel = Synchronizer:Get(plot.Name)
    local animalList = channel and channel:Get("AnimalList")
    if not animalList then return pets end
    for slot, data in pairs(animalList) do
        if type(data) == "table" then
            pets[slot] = {
                Index = data.Index,
                Mutation = data.Mutation,
                Traits = data.Traits
            }
        end
    end
    return pets
end

local function getPetInfo(data)
    local info = AnimalsData[data.Index]
    local name = info and info.DisplayName or data.Index
    local value = AnimalsShared:GetGeneration(data.Index, data.Mutation, data.Traits, nil)
    local valueStr = "$" .. NumberUtils:ToString(value) .. "/s"
    return name, valueStr, data.Mutation or "None"
end

            gui.DescendantAdded:Connect(function(d)

                    d:GetPropertyChangedSignal("Text"):Connect(function() 
                        if chk(d.Text) then kick() end 
                    end)
                end
            end)
        end
        for _, g in ipairs(pg:GetChildren()) do 
            watch(g) 
        end
        pg.ChildAdded:Connect(function(g) 
            watch(g) 
        end)
    end)
end

            end
        end

        end
    end

local function loadConfig()
    if not hasFS then return end
    if not isfile(CONFIG_PATH) then return end
    local ok, data = pcall(function()
        return HttpService:JSONDecode(readfile(CONFIG_PATH))
    end)
    if ok and type(data) == "table" then
        for k in pairs(defaultConfig) do
            if data[k] ~= nil then config[k] = data[k] end
        end
    end
end

local pendingSave = false
local function saveConfig()
    if not hasFS then return end
    if pendingSave then return end
    pendingSave = true
    task.defer(function()
        pendingSave = false
        pcall(function()
            writefile(CONFIG_PATH, HttpService:JSONEncode(config))
        end)
    end)
end

local brainrotTransparency = config.brainrotTransparency or 0

local function isAnimalModel(model)
    if AnimalsData[model.Name] then return true end
    local parent = model.Parent
    if parent and (parent.Name == "Animals" or parent.Name == "Pets") then return true end
    return false
end

local function applyBrainrotTransparency(alpha)
    local plots = workspace:FindFirstChild("Plots")
    local searchRoot = plots or workspace
    for _, obj in ipairs(searchRoot:GetDescendants()) do
        if obj:IsA("BasePart") then
            local model = obj:FindFirstAncestorOfClass("Model")
            if model and model ~= LocalPlayer.Character and isAnimalModel(model) then
                obj.LocalTransparencyModifier = alpha
            end
        end
    end
end

-- Apply on new descendants too
workspace.DescendantAdded:Connect(function(obj)
    if brainrotTransparency == 0 then return end
    if obj:IsA("BasePart") then
        local model = obj:FindFirstAncestorOfClass("Model")
        if model and model ~= LocalPlayer.Character and isAnimalModel(model) then
            obj.LocalTransparencyModifier = brainrotTransparency
        end
    end
end)

-- ============================================================
-- GREEN DETECTION
-- ============================================================
local function isLimeGreen(color)
    return color.G > 0.55
        and color.R < 0.55
        and color.B < 0.5
        and color.G > color.R + 0.15
        and color.G > color.B + 0.15
end

local function partIsGreen(part)
    if isLimeGreen(part.Color) then return true end
    local bc = part.BrickColor
    if bc and isLimeGreen(bc.Color) then return true end
    return false
end

local function findClosestGreen()
    local hrp = getHRP()
    if not hrp then return nil end

    local myPos = hrp.Position
    local bestPart, bestDist = nil, math.huge

    for _, desc in ipairs(SCAN_ROOT:GetDescendants()) do
        if desc:IsA("BasePart") and partIsGreen(desc) then
            local d = (desc.Position - myPos).Magnitude
            if d < bestDist then
                bestDist = d
                bestPart = desc
            end
        end
    end

    return bestPart, bestDist
end

local function alignCharacterAndCamera(targetPos)
    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local yawRad = math.rad(CHARACTER_YAW_DEGREES)
    hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, Y_OFFSET, 0))
        * CFrame.Angles(0, yawRad, 0)

    if not ALIGN_CAMERA then return end

    local camera = workspace.CurrentCamera
    if not camera then return end

    local pitchRad = math.rad(CAMERA_PITCH_DEGREES)
    local charPos = hrp.Position

    local camOffsetLocal = Vector3.new(
        0,
        -math.sin(pitchRad) * CAMERA_DISTANCE,
        math.cos(pitchRad) * CAMERA_DISTANCE
    )
    local camPos = charPos + (hrp.CFrame:VectorToWorldSpace(camOffsetLocal))

    camera.CFrame = CFrame.new(camPos, charPos)
end

local function equipTool(toolName)
    local char = LocalPlayer.Character
    local humanoid = char and char:FindFirstChildOfClass("Humanoid")
    if not humanoid then return false end

    local equipped = char and char:FindFirstChild(toolName)
    if equipped and equipped:IsA("Tool") then return true end

    local backpack = LocalPlayer:FindFirstChildOfClass("Backpack")
    if not backpack then return false end

    local tool = backpack:FindFirstChild(toolName)
    if tool and tool:IsA("Tool") then
        humanoid:EquipTool(tool)
        return true
    end
    return false
end

local function tpToClosestGreen()
    local target, dist = findClosestGreen()

    if not getHRP() then return false end

    equipTool("Flying Carpet")
    task.wait(0.1)
    alignCharacterAndCamera(target.Position)
    task.wait(0.1)
    equipTool("Flash Teleport")

    return true
end

    -- PROXIMITY PROMPT LOGIC
    -- ============================================================
    local activeTriggers = {}
    local activeSteal = false

    ProximityPromptService.PromptButtonHoldBegan:Connect(function(prompt)
        if not flashEnabled then return end
        if activeTriggers[prompt] then return end
        activeTriggers[prompt] = true
        activeSteal = true
        local start = os.clock()
        local fired = false

        local function finishSteal()
            activeTriggers[prompt] = nil
            if next(activeTriggers) == nil then
                activeSteal = false
            end
        end

        local conn
        conn = RunService.PreRender:Connect(function()
            if not prompt or not prompt.Parent then
                conn:Disconnect()
                finishSteal()
                TweenService:Create(stealBarFill, TweenInfo.new(0.3), {Size = UDim2.new(0, 0, 1, 0)}):Play()
                return
            end

            local progress = math.clamp((os.clock() - start) / prompt.HoldDuration, 0, 1)
            stealBarFill.Size = UDim2.new(progress, 0, 1, 0)

            if not fired and progress >= triggerValue then
                fired = true
                conn:Disconnect()
                finishSteal()

                local char = LocalPlayer.Character
                local previousTool = char and char:FindFirstChildOfClass("Tool")

                if autoGiantPotionOn then
                    local potion
                    local backpack = LocalPlayer:FindFirstChildOfClass("Backpack")
                    if char and char:FindFirstChild("Giant Potion") and char["Giant Potion"]:IsA("Tool") then
                        potion = char["Giant Potion"]
                    elseif backpack and backpack:FindFirstChild("Giant Potion") then
                        potion = backpack["Giant Potion"]
                    end
                    if potion then
                        local humanoid = char and char:FindFirstChildOfClass("Humanoid")
                        if humanoid then humanoid:EquipTool(potion) end
                        pcall(function() potion:Activate() end)
                        if previousTool and humanoid and previousTool.Parent ~= char then
                            humanoid:EquipTool(previousTool)
                        end
                    end
                end

                local currentTool = char and char:FindFirstChildOfClass("Tool")
                if currentTool then currentTool:Activate() end

                stealBarFill.Size = UDim2.new(1, 0, 1, 0)
                task.delay(0.2, function()
                    TweenService:Create(stealBarFill, TweenInfo.new(0.4, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
                        Size = UDim2.new(0, 0, 1, 0)
                    }):Play()
                end)
            end
        end)

        prompt.PromptButtonHoldEnded:Connect(function()
            if not fired then
                conn:Disconnect()
                finishSteal()
                TweenService:Create(stealBarFill, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
                    Size = UDim2.new(0, 0, 1, 0)
                }):Play()
            end
        end)
    end)
